} } async function exportSelectedSnapshot() { const snaps = getSnapshotsForCurrentFile(); const selected = getSelectedSnapshot(snaps); if (!selected) { Moonskai.showStatus?.("Select a snapshot to export"); all.push(snap); newestImportedId = snap.id; } trimSnapshots(all); saveAllSnapshots(all); selectedSnapshotId = newestImportedId; renderPanel(true); Moonskai.showStatus?.("Snapshot(s) imported"); } catch (_) { alert("Could not import snapshot file."); "flex" : "none"}; flex-direction:column; overflow:hidden; font-family:system-ui; resize: both; `; panel.innerHTML = `
⏱️ Time Machine
Current file

`; document.body.appendChild(panel); const header = panel.querySelector("#mskTmHeader"); let dragging = false; let offsetX = 0; let offsetY = 0; header.addEventListener("mousedown", (e) => { const target = e.target; if (target && target.closest && target.closest("button")) return; const rect = panel.getBoundingClientRect(); dragging = true; offsetX = e.clientX - rect.left; offsetY = e.clientY - rect.top; panel.style.left = `${Math.round(rect.left)}px`; panel.style.top = `${Math.round(rect.top)}px`; try { e.preventDefault(); } catch (_) {} }, true); document.addEventListener("mousemove", (e) => { if (!dragging) return; panel.style.left = `${Math.max(0, e.clientX - offsetX)}px`; panel.style.top = `${Math.max(0, e.clientY - offsetY)}px`; }, true); document.addEventListener("mouseup", () => { if (!dragging) return; dragging = false; clampPanelToViewport(); }, true); const resizeObserver = new ResizeObserver(() => savePanelState()); resizeObserver.observe(panel); window.addEventListener("resize", clampPanelToViewport); panel.querySelector("#mskTmClose").onclick = () => { panel.style.display = "none"; savePanelState(); }; panel.querySelector("#mskTmCreate").onclick = () => { const ok = createSnapshot(); renderPanel(true); Moonskai.showStatus?.(ok ?
